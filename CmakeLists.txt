cmake_minimum_required(VERSION 3.10)

project(mini-pull)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)


find_package(CURL REQUIRED)
find_package(nlohmann_json CONFIG REQUIRED)


add_executable(my_app src/cli/main.cpp src/http/http_client.cpp src/client/client.cpp)

target_link_libraries(my_app PRIVATE CURL::libcurl)
target_link_libraries(my_app PRIVATE nlohmann_json::nlohmann_json)

target_include_directories(my_app PRIVATE ${CURL_INCLUDE_DIRS})
target_include_directories(my_app PRIVATE /usr/include)
